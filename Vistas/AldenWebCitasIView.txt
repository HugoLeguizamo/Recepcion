CREATE VIEW [dbo].[AldenWebCitasI]
AS
select  V.MovID,V.HoraRequerida as 'Hora',V.Agente,ISNULL(V.ServicioPlacas,'') as 'Placas',ISNULL(UPPER(CONVERT(varchar(200),V.Comentarios)),'') as 'Comentarios'
,C.Nombre as 'NombreCliente',A.Nombre as 'NombreAsesor',v.Estatus,v.Situacion
from Venta V 
left join Venta V1 on V.Mov=V1.Origen and V.MovID=V1.OrigenID
inner join Cte C on V.cliente=C.cliente
inner join Agente A on a.Agente=V.Agente
where V.Mov='Cita Servicio' and V.FechaRequerida=CONVERT(date,GETDATE()) 
--and v1.OrigenID is null



GO
