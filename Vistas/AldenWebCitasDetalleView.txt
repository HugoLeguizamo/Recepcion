

CREATE VIEW [dbo].[AldenWebCitasDetalle]
AS
SELECT
Venta.Mov,
Venta.MovID,
Venta.FechaEmision,
Venta.Moneda,
Venta.Usuario,
ISNULL(Venta.Referencia,'') AS 'Referencia',
ISNULL(Venta.Observaciones,'') as 'Observaciones',
Venta.Estatus,
Venta.FechaRequerida,
Venta.HoraRequerida,
ISNULL(Venta.ServicioTipoOperacion,'') as 'ServicioTipoOperacion',
ISNULL(Venta.ServicioSerie,'') as 'ServicioSerie',
ISNULL(Venta.ServicioPlacas,'') as 'Placas',
ISNULL(Venta.Atencion,'') as 'Atencion',
ISNULL(Venta.AtencionTelefono,'') as 'AtencionTelefono',
ISNULL(Venta.Comentarios,'')as 'Comentarios',
Cte.Cliente,
Cte.Nombre as 'NombreCliente' ,
Cte.Direccion,
Cte.RFC,
ISNULL(Cte.Telefonos,'') as 'Telefonos',
Agente.Agente,
Agente.Nombre as 'NombreAsesor'
FROM Venta
JOIN Cte ON Venta.Cliente=Cte.Cliente
LEFT OUTER JOIN Agente ON Venta.Agente=Agente.Agente
where Mov='Cita Servicio'
GO
