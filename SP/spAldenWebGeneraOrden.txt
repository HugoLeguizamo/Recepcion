
CREATE PROCEDURE [dbo].[spAldenWebGeneraOrden]
@MovID VARCHAR(30)
AS
BEGIN
DECLARE @ID INT,
		@IDSer INT




			SELECT @ID=ID FROM Venta WHERE Mov='Cita Servicio' AND MovID=@MovID

			SELECT @IDSer=ID FROM VENTA WHERE Origen='Cita Servicio' AND OrigenID=@MovID

			INSERT INTO temporalAldenWeb (movid,id,km)values(@MovID,@ID,NULL)

			IF @IDSer IS NULL
			BEGIN
				EXEC spAfectar 'VTAS', @ID, 'GENERAR', 'Todo', 'Servicio', 'CITAS', @Estacion=71		
			END
		
	
	--EXEC spAfectar 'VTAS', @ID, 'GENERAR', 'Todo', 'Servicio', 'HLEGUIZAMO', @Estacion=71	
	--EXEC spEnvioNotificacionRecepcion @MovID

END
